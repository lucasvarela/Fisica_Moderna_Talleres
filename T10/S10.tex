	\section{Normalización y valores esperados en 1D}
	
	Se tiene la siguiente función de onda para un partícula:
	
	\begin{equation*}
	\psi(x,t) = A  \exp\left( -\frac{(x-x_0)^2}{4a^2} + \frac{ip_0 x}{\hbar} + i\omega_0 t    \right)
	\end{equation*}	
	
	\begin{enumerate}
		\item Calcule la constante de normalización $A$.
		\item Calcule el valor esperado de la posición.
		\item Calcule el valor esperado del momento.
	\end{enumerate}	

\begin{center}
	\textbf{SOLUCIÓN}
\end{center}


1.1 La condición de normalización es:

\begin{equation}
\int_{\mathbb{R}} \psi^*(x,t) \psi(x,t) dx = 1
\end{equation}

Se calcula el conjugado complejo de $\psi$:

\begin{equation}
\psi(x,t)^* = A   \exp\left( -\frac{(x-x_0)^2}{4a^2} - \frac{ip_0 x}{\hbar} - i\omega_0 t    \right)
\end{equation}

Por lo que:
\begin{equation}
\psi^*(x,t) \psi(x,t) =A^2  \exp\left( -\frac{(x-x_0)^2}{2a^2} \right) 
\end{equation}

Se inserta la expresión anterior en la condición de normalización. Luego se hace el cambio de variable $u = (x-x_0)/\sqrt{2}a$ y por tanto el diferencial transforma como $du = dx/\sqrt{2}a$. El resultado es:
\begin{equation}
\int_{\mathbb{R}}  A^2  \exp\left( -\frac{(x-x_0)^2}{2a^2} \right) dx = A^2 \sqrt{2}a   \int_{\mathbb{R}}  \e^{-u^2} du
 \end{equation}

La integral que queda es conocida (integral gaussiana) y da $\sqrt{\pi}$. Por lo que la constante de normalización es:

\begin{equation}
A^2 = \frac{1}{a\sqrt{2\pi}}
\end{equation}
1.2 Para calcular el valor esperado se calcula la siguiente integral:

\begin{equation}
\braket{x} = \int_{\mathbb{R}} \psi^*(x,t) x \psi(x,t) dx = \frac{1}{a\sqrt{2\pi}}\int_{\mathbb{R}}   x \exp\left( -\frac{(x-x_0)^2}{2a^2} \right) dx 
\end{equation}

Para ello se hace nuevamente el cambio de variable $u = (x-x_0)/\sqrt{2}a$, de donde se despeja $x =\sqrt{2} a u + x_0$. La integral es:

\begin{equation}
 \frac{1}{a\sqrt{2\pi}}\int_{\mathbb{R}}   x \exp\left( -\frac{(x-x_0)^2}{2a^2} \right) dx  = \frac{ \sqrt{2}a}{a\sqrt{2\pi}}  \int_{\mathbb{R}}  (\sqrt{2} a u + x_0)\e^{-u^2} du
\end{equation}

La integral $ \int u \e^{-u^2} du$ es de una función impar por lo que es cero. La integral que falta es la guassiana por lo que:

\begin{equation}
\braket{x} = x_0
\end{equation}

1.3 


\begin{equation}
\braket{p} =  \int_{\mathbb{R}}  \psi^*(x,t) p \psi(x,t) dx = -i\hbar \int_{\mathbb{R}}  \psi^*(x,t) \frac{d}{dx} \psi(x,t) dx
\end{equation}

\begin{equation}
\frac{d}{dx} \psi(x,t)   =     A \left( -\frac{x-x_0}{2a^2} - \frac{ip_0 }{\hbar}  \right)  \exp\left( -\frac{(x-x_0)^2}{4a^2} - \frac{ip_0 x}{\hbar} - i\omega_0 t    \right)
\end{equation}


\begin{equation}
\braket{p} = - \frac{i\hbar}{a\sqrt{2\pi}} \int_{\mathbb{R}} dx\; \left( -\frac{x-x_0}{2a^2} - \frac{ip_0 }{\hbar}  \right)\exp\left( -\frac{(x-x_0)^2}{2a^2} \right)
\end{equation}

Ambas integrales se hacen con la misma sustitución $u$ que se usó  en los incisos anteriores. La primera integral da cero y la segunda es una guassiana. 

\begin{align}
\int_{\mathbb{R}} dx\;  \frac{x-x_0}{2a^2} \exp\left( -\frac{(x-x_0)^2}{2a^2} \right) &= 0\\
\int_{\mathbb{R}} dx\; \exp\left( -\frac{(x-x_0)^2}{2a^2} \right) &= a\sqrt{2 \pi}
\end{align}

Con los resultado anteriores se tiene:

\begin{equation}
\braket{p} =  p_0
\end{equation}

	\section{Valor esperado en 3D}
	
	Una electrón está sujeto a la siguiente energía potencial:
	
	\begin{equation}
	V(\vec{r} ) = \frac{e^2}{r}
	\end{equation}
	
	Donde $ r = |\vec{r}|$. En su estado base el electrón tiene la siguiente función de onda definida sobre todo $\mathbb{R}^3$:
	
	\begin{equation*}
	\psi(\vec{r}) = \frac{1}{\sqrt{\pi a^3}} \e^{-\frac{r}{a}}
	\end{equation*}
	
	Donde $a = \hbar^2/(m_e e^2)$. Calcule $\braket{V}$, el valor esperado de la energía potencial en el estado base.  
	
	
	\begin{center}
		\textbf{SOLUCIÓN}
	\end{center}

El valor esperado de $V$ se obtiene calculando la siguiente integral:

\begin{equation}
\braket{V} = \int_{\mathbb{R}^3} \psi^*(\vec{r}) V(\vec{r}) \psi(\vec{r}) d^3 r
\end{equation}

Es conveniente realizar esta integral en coordenadas esféricas. El diferencial de volumen en coordenadas esféricas es $d^3 r = r^2\sin \theta d\theta d\varphi dr$, donde $\theta$ es el ángulo polar y $\varphi$ el azimutal. Con esto:

\begin{equation}
 \int_{\mathbb{R}^3} \psi^*(\vec{r}) V(\vec{r}) \psi(\vec{r}) d^3 r = \frac{e^2}{\pi a^3}\int_{0}^{2\pi}d\varphi \int_{0}^{\pi} \sin\theta \;d\theta \int_{0}^{\infty} r \e^{-2r/a} \; dr
\end{equation}
 
 Las primeras dos integrales son básicas y dan en total $4\pi$. La integral radial se hace por método de sustitución. El resultado es:
 
 \begin{equation}
 \braket{V} = - \frac{e^2}{a} = - \frac{m_e e^2}{\hbar^2}
 \end{equation}

	\section{Valor esperado del momento}
	
	Una partícula atrapada en un pozo de potencial infinito tiene la siguiente función de onda:
	
	\begin{equation}
	\psi(x) =\begin{cases}
	\sqrt{\frac{2}{L}} \sin\left(\frac{2\pi x}{L}\right) & x\in[0,L]\\
	\quad 0 & x\notin[0,L]
	\end{cases}
	\end{equation}
	
	
	\begin{enumerate}
		\item Calcule $\Delta p$.
		\item Calcule $\Delta x$.
		\item Calcule $\Delta x\Delta p$. ¿Se cumple el principio de incertidumbre $\Delta x\Delta p \geq \hbar/2$?.
	\end{enumerate}
	
	
	\begin{center}
		\textbf{SOLUCIÓN}
	\end{center}

3.1 La desviación estándar se calcula con la siguiente relación:

\begin{equation}
(\Delta p)^2 = \braket{p^2} - \braket{p}^2
 \end{equation} 

Entonces lo que se debe hacer es calcular las siguientes integrales:

\begin{align}
\braket{p^2} &=  \int_{\mathbb{R}}  \psi^*(x) p^2 \psi(x) dx = -\hbar^2\int_{0}^L  \psi^*(x) \frac{d^2}{dx^2} \psi(x) dx \\
\braket{p} &=  \int_{\mathbb{R}}  \psi^*(x) p \psi(x) dx = -i\hbar \int_{0}^L  \psi^*(x) \frac{d}{dx} \psi(x) dx
\end{align}

donde se usó  que $p = -i\hbar \frac{d}{dx}$. Estas dan:

\begin{align}
\braket{p^2} &=    \left(\frac{2\pi \hbar}{L}\right)^2  \left( \frac{2}{L}\int_{\mathbb{R}}  dx\;  \sin^2\left(\frac{2\pi x}{L}\right)\right)\\
\braket{p} &=  -\frac{4 \pi i \hbar}{L}\int_{\mathbb{R}} dx\;  \sin\left(\frac{2\pi x}{L}\right)  \cos\left(\frac{2\pi x}{L}\right)
\end{align}

Las integrales que quedan se pueden hacer por inspección. La primera entre parentesis esta dada por la condición de normalización y la segunda al integral $\sin(\alpha x) \cos(\alpha x)$ sobre un periodo es cero. Por lo que:

\begin{align}
\braket{p^2} &=    \left(\frac{2\pi \hbar}{L}\right)^2  \\
\braket{p} &= 0
\end{align}	

Por lo que la desviación estándar es:

\begin{equation}
\Delta p = \frac{2\pi \hbar}{L}
\end{equation} 


3.2 La desviación estándar se calcula con la siguiente relación:

\begin{equation}
(\Delta x)^2 = \braket{x^2} - \braket{x}^2
\end{equation} 

Entonces lo que se debe hacer es calcular las siguientes integrales:

\begin{align}
\braket{x^2} &=  \int_{\mathbb{R}}  \psi^*(x) x^2 \psi(x) dx =  \int_{0}^L \psi^*(x) x^2 \psi(x) dx \\
\braket{x} &=  \int_{\mathbb{R}}  \psi^*(x) x \psi(x) dx =  \int_{0}^L \psi^*(x) x \psi(x) dx
\end{align}

Estas dan:

\begin{align}
\braket{x^2} &=  \frac{2}{L}\int_{\mathbb{R}}  dx\; x^2 \sin^2\left(\frac{2\pi x}{L}\right)\\
\braket{x} &= \frac{2}{L}\int_{\mathbb{R}}  dx\;  x\sin^2\left(\frac{2\pi x}{L}\right)
\end{align}

Las integrales se calculan integrando por partes:
\begin{align}
\braket{x^2} &=  \frac{2}{L} \frac{(8\pi^2-3)L^3}{48 \pi^2} = L^2 \left(  \frac{1}{3} - \frac{1}{8\pi^2}\right)   \\
\braket{x} &= \frac{2}{L} \frac{L^2}{4} = \frac{L}{2}
\end{align}

Con esto la desviación estándar es:
\begin{equation}
\Delta x =  \frac{L}{2} \sqrt{ \frac{1}{3} - \frac{1}{2\pi^2}}
\end{equation}

3.3 El producto de las incertidumbres es:


\begin{equation}
\Delta x  \Delta p = \pi \hbar \sqrt{ \frac{1}{3} - \frac{1}{2\pi^2}} 
\end{equation}

El principio de incertidumbre de Heisenberg ya que el producto de las varianzas es mayor a $\hbar/2$:
\begin{equation}
\Delta x  \Delta p \approx \frac{\pi \hbar }{2} > \frac{\hbar}{2}
\end{equation}
